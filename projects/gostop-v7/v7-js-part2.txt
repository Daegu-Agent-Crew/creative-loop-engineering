// === MAIN GAME ===
class GoStopGame {
  constructor(isMultiplayer = false, myPlayerIndex = 0, isHost = true) {
    this.themeManager = themeManager;
    this.isMultiplayer = isMultiplayer;
    this.myPlayerIndex = myPlayerIndex;
    this.isHost = isHost;
    this.newGame();
  }

  toggleTheme() { this.themeManager.toggle(); }

  newGame() {
    this.deck = shuffle([...ALL_CARDS]);
    this.floor = [];
    this.ppukStacks = {};
    this.players = [
      { name: '나', hand: [], captured: { gwang:[], tti:[], dan:[], pi:[] }, goCount: 0, score: 0, heundeulgi: false, heundeulgiMonths: [], ppukCount: 0, bombs: 0 },
      { name: this.isMultiplayer ? '상대' : 'CPU', hand: [], captured: { gwang:[], tti:[], dan:[], pi:[] }, goCount: 0, score: 0, heundeulgi: false, heundeulgiMonths: [], ppukCount: 0, bombs: 0 },
    ];
    this.currentPlayer = 0;
    this.turnCount = 0;
    this.gameOver = false;
    this.waitingForInput = false;
    this.phase = 'deal';
    this.pendingCapture = null;

    this.deal();
  }

  deal() {
    for (let i = 0; i < 10; i++) {
      this.players[0].hand.push(this.deck.pop());
      this.players[1].hand.push(this.deck.pop());
    }
    for (let i = 0; i < 8; i++) {
      this.floor.push(this.deck.pop());
    }

    this.players[0].hand.sort((a, b) => a.month - b.month || a.id - b.id);
    this.players[1].hand.sort((a, b) => a.month - b.month || a.id - b.id);

    // In multiplayer, host sends game state
    if (this.isMultiplayer && this.isHost) {
      mpClient.send({
        type: 'game_state',
        action: 'init',
        deck: this.deck.map(c => c.id),
        floor: this.floor.map(c => c.id),
        p0hand: this.players[0].hand.map(c => c.id),
        p1hand: this.players[1].hand.map(c => c.id),
      });
    }

    this.render();
    this.checkInitialSpecials();
  }

  restoreFromSave(data) {
    this.deck = SaveManager.restoreCards(data.deck);
    this.floor = SaveManager.restoreCards(data.floor);
    this.ppukStacks = data.ppukStacks || {};
    this.turnCount = data.turnCount || 0;
    this.currentPlayer = data.currentPlayer || 0;
    this.gameOver = false;

    data.players.forEach((pd, i) => {
      this.players[i].hand = SaveManager.restoreCards(pd.hand);
      this.players[i].captured = {
        gwang: SaveManager.restoreCards(pd.captured.gwang),
        tti: SaveManager.restoreCards(pd.captured.tti),
        dan: SaveManager.restoreCards(pd.captured.dan),
        pi: SaveManager.restoreCards(pd.captured.pi),
      };
      this.players[i].goCount = pd.goCount;
      this.players[i].score = pd.score;
      this.players[i].heundeulgi = pd.heundeulgi;
      this.players[i].heundeulgiMonths = pd.heundeulgiMonths;
      this.players[i].ppukCount = pd.ppukCount;
      this.players[i].bombs = pd.bombs;
    });

    this.phase = 'play';
    this.render();
    NotificationManager.show('💾 불러오기 완료!', '', 'success', 800);

    if (this.currentPlayer === 1) {
      setTimeout(() => this.cpuPlay(), 800);
    } else {
      this.checkPlayerBomb();
    }
  }

  handleRemoteAction(msg) {
    // Handle actions from remote player in multiplayer
    switch(msg.type) {
      case 'game_state':
        if (msg.action === 'init' && !this.isHost) {
          this.deck = SaveManager.restoreCards(msg.deck);
          this.floor = SaveManager.restoreCards(msg.floor);
          this.players[0].hand = SaveManager.restoreCards(msg.p0hand);
          this.players[1].hand = SaveManager.restoreCards(msg.p1hand);
          // Swap perspective for non-host
          if (this.myPlayerIndex === 1) {
            const tmp = this.players[0];
            this.players[0] = this.players[1];
            this.players[1] = tmp;
          }
          this.players[0].hand.sort((a, b) => a.month - b.month || a.id - b.id);
          this.players[1].hand.sort((a, b) => a.month - b.month || a.id - b.id);
          this.render();
          this.checkInitialSpecials();
        }
        break;
      case 'play_card':
        // Opponent played a card - execute it
        this.executePlayRemote(msg.cardId);
        break;
    }
  }

  async executePlayRemote(cardId) {
    const card = this.players[1].hand.find(c => c.id === cardId);
    if (!card) return;
    await this.executePlay(1, card);
  }

  async checkInitialSpecials() {
    for (let p = 0; p < 2; p++) {
      const hand = this.players[p].hand;
      for (let m = 1; m <= 12; m++) {
        if (hand.filter(c => c.month === m).length === 4) {
          await NotificationManager.showChongtong();
          this.players[p].score = 10;
          this.endGame(p, '총통');
          return;
        }
      }
    }

    const playerHand = this.players[0].hand;
    const heundeulgiMonths = [];
    for (let m = 1; m <= 12; m++) {
      if (playerHand.filter(c => c.month === m).length >= 3) {
        heundeulgiMonths.push(m);
      }
    }
    if (heundeulgiMonths.length > 0) {
      this.showHeundeulgiChoice(heundeulgiMonths);
      return;
    }

    if (!this.isMultiplayer) this.checkCpuHeundeulgi();
    this.phase = 'play';
    this.startTurn();
  }

  // === ENHANCED CPU HEUNDEULGI (v7) ===
  checkCpuHeundeulgi() {
    const cpuHand = this.players[1].hand;
    for (let m = 1; m <= 12; m++) {
      if (cpuHand.filter(c => c.month === m).length >= 3) {
        if (aiDifficulty === 'easy') {
          // Easy AI doesn't use heundeulgi
          continue;
        }
        if (aiDifficulty === 'medium' && Math.random() < 0.5) {
          continue; // 50% chance to skip
        }
        this.players[1].heundeulgi = true;
        this.players[1].heundeulgiMonths.push(m);
      }
    }
    if (this.players[1].heundeulgi) {
      NotificationManager.show('🤖 흔들기!', 'CPU가 흔들기 선언', 'info');
    }
  }

  showHeundeulgiChoice(months) {
    const monthNames = months.map(m => `${MONTH_NAMES[m]}(${MONTH_FLOWERS[m]})`).join(', ');
    const modal = document.getElementById('modal-container');
    modal.innerHTML = `
      <div class="modal-overlay">
        <div class="modal">
          <h2>👋 흔들기</h2>
          <p>${monthNames} - 같은 월 3장 보유!<br>흔들기를 선언하시겠습니까?<br>(승리시 점수 2배)</p>
          <button class="btn btn-success" onclick="game.declareHeundeulgi(${JSON.stringify(months)})">흔들기!</button>
          <button class="btn btn-secondary" onclick="game.declineHeundeulgi()">안함</button>
        </div>
      </div>`;
  }

  declareHeundeulgi(months) {
    this.players[0].heundeulgi = true;
    this.players[0].heundeulgiMonths = months;
    document.getElementById('modal-container').innerHTML = '';
    NotificationManager.showHeundeulgi();

    if (this.isMultiplayer) {
      mpClient.send({ type: 'heundeulgi', months });
    }

    setTimeout(() => {
      document.querySelectorAll('#player-hand .card').forEach(el => {
        const cardMonth = parseInt(el.dataset.month);
        if (months.includes(cardMonth)) {
          el.classList.add('shake-anim');
          setTimeout(() => el.classList.remove('shake-anim'), 500);
        }
      });
    }, 100);

    if (!this.isMultiplayer) this.checkCpuHeundeulgi();
    this.phase = 'play';
    this.render();
    this.startTurn();
  }

  declineHeundeulgi() {
    document.getElementById('modal-container').innerHTML = '';
    if (!this.isMultiplayer) this.checkCpuHeundeulgi();
    this.phase = 'play';
    this.render();
    this.startTurn();
  }

  // === TURN LOGIC ===
  startTurn() {
    if (this.gameOver) return;
    this.turnCount++;
    this.render();

    if (this.currentPlayer === 1) {
      if (this.isMultiplayer) {
        // Wait for remote player
        this.waitingForInput = false;
        this.render();
      } else {
        setTimeout(() => this.cpuPlay(), 800);
      }
    } else {
      this.checkPlayerBomb();
    }
  }

  checkPlayerBomb() {
    const hand = this.players[0].hand;
    const bombMonths = [];
    for (let m = 1; m <= 12; m++) {
      const inHand = hand.filter(c => c.month === m).length;
      const onFloor = this.floor.filter(c => c.month === m).length;
      if (inHand >= 3 && onFloor >= 1) {
        bombMonths.push(m);
      }
    }

    if (bombMonths.length > 0) {
      this.showBombChoice(bombMonths);
    } else {
      this.waitingForInput = true;
      this.render();
    }
  }

  showBombChoice(months) {
    const btns = document.getElementById('action-buttons');
    let html = '<span style="font-size:0.8em;opacity:0.7">폭탄 가능: </span>';
    months.forEach(m => {
      html += `<button class="btn btn-warning btn-small" onclick="game.executeBomb(${m})">💣 ${MONTH_NAMES[m]}</button>`;
    });
    html += `<button class="btn btn-secondary btn-small" onclick="game.skipBomb()">패스</button>`;
    btns.innerHTML = html;
    this.waitingForInput = true;
    this.render();
  }

  async executeBomb(month) {
    const p = this.players[this.currentPlayer];
    const handCards = p.hand.filter(c => c.month === month);
    const floorCards = this.floor.filter(c => c.month === month);
    const allCards = [...handCards, ...floorCards];

    p.hand = p.hand.filter(c => c.month !== month);
    this.floor = this.floor.filter(c => c.month !== month);

    allCards.forEach(c => this.captureCard(this.currentPlayer, c));
    p.bombs++;

    this.stealPi(this.currentPlayer, 1 - this.currentPlayer, 1);

    if (this.isMultiplayer) {
      mpClient.send({ type: 'bomb_action', month });
    }

    await NotificationManager.showBomb();
    document.getElementById('action-buttons').innerHTML = '';

    this.render();
    this.waitingForInput = true;
    this.render();
  }

  skipBomb() {
    document.getElementById('action-buttons').innerHTML = '';
    this.waitingForInput = true;
    this.render();
  }

  async playCard(cardId) {
    if (!this.waitingForInput || this.currentPlayer !== 0) return;
    this.waitingForInput = false;

    const card = this.players[0].hand.find(c => c.id === cardId);
    if (!card) return;

    soundManager.cardPlace();

    if (this.isMultiplayer) {
      mpClient.send({ type: 'play_card', cardId: card.id });
    }

    await this.executePlay(0, card);
  }

  async executePlay(playerIdx, card) {
    const player = this.players[playerIdx];
    player.hand = player.hand.filter(c => c.id !== card.id);

    const matches = this.floor.filter(c => c.month === card.month);

    if (matches.length === 0) {
      this.floor.push(card);
      this.render();
      await delay(400);
      await this.drawFromDeck(playerIdx);
    } else if (matches.length === 1) {
      this.floor = this.floor.filter(c => c.id !== matches[0].id);
      this.pendingCapture = [card, matches[0]];
      soundManager.capture();
      this.render();
      await delay(400);
      await this.drawFromDeck(playerIdx);
    } else if (matches.length === 2) {
      if (playerIdx === 0 && !this.isMultiplayer || (this.isMultiplayer && playerIdx === 0)) {
        this.showMatchChoice(card, matches);
        return;
      } else {
        const pick = matches[0];
        this.floor = this.floor.filter(c => c.id !== pick.id);
        this.pendingCapture = [card, pick];
        this.render();
        await delay(400);
        await this.drawFromDeck(playerIdx);
      }
    } else if (matches.length === 3) {
      const ppukKey = card.month;
      if (this.ppukStacks[ppukKey]) {
        const ppukCreator = this.ppukStacks[ppukKey].creator;
        if (ppukCreator === playerIdx) {
          await NotificationManager.showJabppuk();
          this.stealPi(1 - playerIdx, playerIdx, 2);
        }
        delete this.ppukStacks[ppukKey];
      }
      const all = this.floor.filter(c => c.month === card.month);
      this.floor = this.floor.filter(c => c.month !== card.month);
      this.pendingCapture = [card, ...all];
      soundManager.capture();
      this.render();
      await delay(400);
      await this.drawFromDeck(playerIdx);
    }
  }

  showMatchChoice(playedCard, matches) {
    const modal = document.getElementById('modal-container');
    let html = `<div class="modal-overlay"><div class="modal"><h2>카드 선택</h2><p>${MONTH_NAMES[playedCard.month]} 중 가져갈 카드를 선택하세요</p><div class="match-options">`;
    matches.forEach(m => {
      html += `<div class="match-option" onclick="game.selectMatch(${playedCard.id}, ${m.id})">${m.icon} ${m.name}</div>`;
    });
    html += `</div></div></div>`;
    modal.innerHTML = html;
  }

  async selectMatch(playedCardId, matchCardId) {
    document.getElementById('modal-container').innerHTML = '';
    const playedCard = ALL_CARDS.find(c => c.id === playedCardId);
    const matchCard = this.floor.find(c => c.id === matchCardId);
    this.floor = this.floor.filter(c => c.id !== matchCardId);
    this.pendingCapture = [playedCard, matchCard];
    soundManager.capture();

    if (this.isMultiplayer) {
      mpClient.send({ type: 'match_choice', playedCardId, matchCardId });
    }

    this.render();
    await delay(400);
    await this.drawFromDeck(0);
  }

  async drawFromDeck(playerIdx) {
    if (this.deck.length === 0) {
      if (this.pendingCapture) {
        this.pendingCapture.forEach(c => this.captureCard(playerIdx, c));
        this.pendingCapture = null;
      }
      this.render();
      this.checkEndOfGame();
      return;
    }

    const drawn = this.deck.pop();
    soundManager.cardFlip();
    const drawnMatches = this.floor.filter(c => c.month === drawn.month);

    if (drawnMatches.length === 0) {
      this.floor.push(drawn);
      if (this.pendingCapture) {
        this.pendingCapture.forEach(c => this.captureCard(playerIdx, c));
        this.pendingCapture = null;
      }
    } else if (drawnMatches.length === 1) {
      this.floor = this.floor.filter(c => c.id !== drawnMatches[0].id);
      if (this.pendingCapture) {
        this.pendingCapture.forEach(c => this.captureCard(playerIdx, c));
        this.pendingCapture = null;
      }
      this.captureCard(playerIdx, drawn);
      this.captureCard(playerIdx, drawnMatches[0]);
      soundManager.capture();
    } else if (drawnMatches.length === 2) {
      this.floor.push(drawn);
      this.players[playerIdx].ppukCount++;
      const ppukCount = this.players[playerIdx].ppukCount;

      this.ppukStacks[drawn.month] = { creator: playerIdx, count: 3 };

      if (this.turnCount <= 1 && ppukCount === 1) {
        await NotificationManager.showFirstPpuk();
      }

      await NotificationManager.showPpuk(ppukCount);

      if (ppukCount >= 3) {
        await NotificationManager.show('💥 삼연뻑!', '게임 즉시 종료!', 'danger', 1500);
        this.players[playerIdx].score = 7;
        if (this.pendingCapture) {
          this.pendingCapture.forEach(c => this.captureCard(playerIdx, c));
          this.pendingCapture = null;
        }
        this.endGame(playerIdx, '삼연뻑');
        return;
      }

      if (this.pendingCapture) {
        this.pendingCapture.forEach(c => this.captureCard(playerIdx, c));
        this.pendingCapture = null;
      }
    } else if (drawnMatches.length === 3) {
      const all = this.floor.filter(c => c.month === drawn.month);
      this.floor = this.floor.filter(c => c.month !== drawn.month);
      if (this.pendingCapture) {
        this.pendingCapture.forEach(c => this.captureCard(playerIdx, c));
        this.pendingCapture = null;
      }
      [drawn, ...all].forEach(c => this.captureCard(playerIdx, c));
      soundManager.capture();

      if (this.ppukStacks[drawn.month]) {
        const ppukCreator = this.ppukStacks[drawn.month].creator;
        if (ppukCreator === playerIdx) {
          await NotificationManager.showJabppuk();
          this.stealPi(1 - playerIdx, playerIdx, 2);
        } else {
          this.stealPi(playerIdx, ppukCreator, 1);
        }
        delete this.ppukStacks[drawn.month];
      }
    }

    this.render();
    await delay(300);

    const score = this.calculateScore(playerIdx);
    if (score >= 7) {
      this.players[playerIdx].score = score;
      await this.showGoStopChoice(playerIdx);
      return;
    }

    this.endTurn();
  }

  captureCard(playerIdx, card) {
    const p = this.players[playerIdx];
    const type = card.type;
    if (type === 'gwang') p.captured.gwang.push(card);
    else if (type === 'tti') p.captured.tti.push(card);
    else if (type === 'dan') p.captured.dan.push(card);
    else if (type === 'pi' || type === 'ssang-pi') p.captured.pi.push(card);
  }

  stealPi(toPlayer, fromPlayer, count) {
    const from = this.players[fromPlayer].captured.pi;
    const stolen = from.splice(0, Math.min(count, from.length));
    this.players[toPlayer].captured.pi.push(...stolen);
  }

  getPiCount(playerIdx) {
    let count = 0;
    this.players[playerIdx].captured.pi.forEach(c => {
      count += c.type === 'ssang-pi' ? 2 : 1;
    });
    return count;
  }

  calculateScore(playerIdx) {
    const p = this.players[playerIdx];
    let score = 0;

    const gwangCount = p.captured.gwang.length;
    const hasBiGwang = p.captured.gwang.some(c => c.month === 12);
    if (gwangCount === 5) score += 15;
    else if (gwangCount === 4) score += 4;
    else if (gwangCount === 3) {
      score += hasBiGwang ? 2 : 3;
    }

    const ttiCount = p.captured.tti.length;
    if (ttiCount >= 5) score += ttiCount - 4;

    const hongdan = p.captured.tti.filter(c => c.icon === '🎀').length >= 3;
    const chodan = p.captured.tti.filter(c => c.icon === '🟦').length >= 3;
    const cheongdan = p.captured.tti.filter(c => c.icon === '🟥').length >= 3;
    if (hongdan) score += 3;
    if (chodan) score += 3;
    if (cheongdan) score += 3;

    const danCount = p.captured.dan.length;
    if (danCount >= 5) score += danCount - 4;

    const piCount = this.getPiCount(playerIdx);
    if (piCount >= 10) score += piCount - 9;

    return score;
  }

  // === ENHANCED GO/STOP CHOICE (v7) ===
  async showGoStopChoice(playerIdx) {
    if (playerIdx === 1 && !this.isMultiplayer) {
      const score = this.calculateScore(1);
      let shouldStop = false;

      if (aiDifficulty === 'easy') {
        shouldStop = score >= 7; // Always stops at minimum
      } else if (aiDifficulty === 'medium') {
        shouldStop = score >= 10 || this.players[1].goCount >= 2;
      } else {
        // Hard: consider opponent's score and risk
        const opponentScore = this.calculateScore(0);
        const goCount = this.players[1].goCount;
        const remainingCards = this.players[1].hand.length;

        if (score >= 12) shouldStop = true;
        else if (goCount >= 3) shouldStop = true;
        else if (score >= 10 && opponentScore >= 5) shouldStop = true;
        else if (score >= 8 && remainingCards <= 2) shouldStop = true;
        else if (goCount >= 2 && opponentScore >= 7) shouldStop = true;
        else shouldStop = false;
      }

      if (shouldStop) {
        this.players[1].score = score;
        this.endGame(1, '스톱');
      } else {
        this.players[1].goCount++;
        await NotificationManager.showGo(this.players[1].goCount);
        this.endTurn();
      }
      return;
    }

    if (this.isMultiplayer && playerIdx === 1) {
      // Wait for remote decision
      return;
    }

    const score = this.calculateScore(0);
    const goCount = this.players[0].goCount;
    const modal = document.getElementById('modal-container');
    modal.innerHTML = `
      <div class="modal-overlay">
        <div class="modal">
          <h2 class="go-pulse">🎯 ${score}점!</h2>
          <p>현재 ${goCount}고 상태<br>고를 외치면 점수가 올라갑니다!</p>
          <button class="btn btn-warning" onclick="game.declareGo()">고! (계속)</button>
          <button class="btn btn-success" onclick="game.declareStop()">스톱! (승리)</button>
        </div>
      </div>`;
  }

  async declareGo() {
    document.getElementById('modal-container').innerHTML = '';
    this.players[0].goCount++;

    if (this.isMultiplayer) {
      mpClient.send({ type: 'go_stop', action: 'go' });
    }

    await NotificationManager.showGo(this.players[0].goCount);
    this.endTurn();
  }

  declareStop() {
    document.getElementById('modal-container').innerHTML = '';
    this.players[0].score = this.calculateScore(0);

    if (this.isMultiplayer) {
      mpClient.send({ type: 'go_stop', action: 'stop' });
    }

    this.endGame(0, '스톱');
  }

  endTurn() {
    if (this.gameOver) return;

    if (this.players[0].hand.length === 0 && this.players[1].hand.length === 0) {
      this.checkEndOfGame();
      return;
    }

    this.currentPlayer = 1 - this.currentPlayer;
    this.startTurn();
  }

  checkEndOfGame() {
    const s0 = this.calculateScore(0);
    const s1 = this.calculateScore(1);
    this.players[0].score = s0;
    this.players[1].score = s1;

    if (s0 > s1) this.endGame(0, '점수');
    else if (s1 > s0) this.endGame(1, '점수');
    else this.endGame(-1, '무승부');
  }

  // === ENHANCED CPU AI (v7) ===
  async cpuPlay() {
    if (this.gameOver) return;
    const cpu = this.players[1];
    if (cpu.hand.length === 0) { this.endTurn(); return; }

    // Check for bomb (all difficulties)
    for (let m = 1; m <= 12; m++) {
      const inHand = cpu.hand.filter(c => c.month === m).length;
      const onFloor = this.floor.filter(c => c.month === m).length;
      if (inHand >= 3 && onFloor >= 1) {
        const handCards = cpu.hand.filter(c => c.month === m);
        const floorCards = this.floor.filter(c => c.month === m);
        cpu.hand = cpu.hand.filter(c => c.month !== m);
        this.floor = this.floor.filter(c => c.month !== m);
        [...handCards, ...floorCards].forEach(c => this.captureCard(1, c));
        cpu.bombs++;
        this.stealPi(1, 0, 1);
        await NotificationManager.showBomb();
        this.render();
        await delay(500);
      }
    }
    if (cpu.hand.length === 0) { this.endTurn(); return; }

    let bestCard = null;

    if (aiDifficulty === 'easy') {
      // Random play
      bestCard = cpu.hand[Math.floor(Math.random() * cpu.hand.length)];
    } else if (aiDifficulty === 'medium') {
      // Basic strategy: prioritize gwang, then matching, consider chongtong
      bestCard = this.cpuMediumStrategy(cpu);
    } else {
      // Hard: advanced strategy with opponent analysis
      bestCard = this.cpuHardStrategy(cpu);
    }

    if (!bestCard) bestCard = cpu.hand[0];
    await this.executePlay(1, bestCard);
  }

  cpuMediumStrategy(cpu) {
    let bestCard = null;
    let bestScore = -1;

    cpu.hand.forEach(card => {
      const matches = this.floor.filter(c => c.month === card.month);
      let score = 0;
      if (matches.length > 0) {
        score += 10;
        matches.forEach(m => {
          if (m.type === 'gwang') score += 30;
          else if (m.type === 'dan') score += 8;
          else if (m.type === 'tti') score += 6;
          else score += 1;
        });
        if (card.type === 'gwang') score += 30;
        else if (card.type === 'dan') score += 8;
        else if (card.type === 'tti') score += 6;
      } else {
        // Penalty for placing valuable cards on floor
        if (card.type === 'gwang') score -= 15;
        else if (card.type === 'dan') score -= 5;
        else if (card.type === 'tti') score -= 3;
      }
      // Consider chongtong potential
      const sameMonth = cpu.hand.filter(c => c.month === card.month).length;
      if (sameMonth >= 3) score += 5;

      if (score > bestScore) {
        bestScore = score;
        bestCard = card;
      }
    });
    return bestCard;
  }

  cpuHardStrategy(cpu) {
    let bestCard = null;
    let bestScore = -Infinity;

    const playerCaptured = this.players[0].captured;
    const playerGwangCount = playerCaptured.gwang.length;
    const playerTtiCount = playerCaptured.tti.length;
    const cpuGwangCount = cpu.captured.gwang.length;
    const cpuTtiCount = cpu.captured.tti.length;

    cpu.hand.forEach(card => {
      const matches = this.floor.filter(c => c.month === card.month);
      let score = 0;

      if (matches.length > 0) {
        score += 15;
        matches.forEach(m => {
          if (m.type === 'gwang') score += 40;
          else if (m.type === 'dan') score += 10;
          else if (m.type === 'tti') {
            score += 8;
            // Check if completing hongdan/chodan/cheongdan set
            const ttiIcon = m.icon;
            const capturedSame = cpu.captured.tti.filter(c => c.icon === ttiIcon).length;
            if (capturedSame >= 2) score += 20; // completing a set!
          }
          else score += 2;
        });
        if (card.type === 'gwang') {
          score += 40;
          if (cpuGwangCount >= 2) score += 30; // getting closer to gwang bonus
        }
        else if (card.type === 'tti') {
          score += 8;
          const sameIcon = cpu.captured.tti.filter(c => c.icon === card.icon).length;
          if (sameIcon >= 2) score += 20;
        }
        else if (card.type === 'dan') score += 10;
      } else {
        // No match - card goes to floor
        if (card.type === 'gwang') score -= 25;
        else if (card.type === 'dan') score -= 8;
        else if (card.type === 'tti') score -= 5;
        else score += 1; // pi cards are OK to place

        // Defensive: don't feed opponent gwang
        if (card.type === 'gwang' && playerGwangCount >= 2) score -= 15;
      }

      // Strategic: avoid creating ppuk situations for opponent
      const floorSameMonth = this.floor.filter(c => c.month === card.month).length;
      if (matches.length === 0 && floorSameMonth === 1) {
        // Creates a pair on floor - potential ppuk risk
        score -= 3;
      }

      // Prefer cards where we hold another of same month (can capture later)
      const sameMonthInHand = cpu.hand.filter(c => c.month === card.month && c.id !== card.id).length;
      if (matches.length === 0 && sameMonthInHand >= 1) {
        score += 5; // we can capture it next turn
      }

      if (score > bestScore) {
        bestScore = score;
        bestCard = card;
      }
    });
    return bestCard;
  }

  // === GAME END ===
  async endGame(winnerIdx, reason) {
    this.gameOver = true;

    if (winnerIdx === -1) {
      this.recordStats(-1, 0, reason);
      this.showGameOver('무승부', 0, 0, {}, reason);
      return;
    }

    const winner = this.players[winnerIdx];
    const loser = this.players[1 - winnerIdx];
    let baseScore = winner.score;
    let multiplier = 1;
    const bonuses = {};

    if (winner.heundeulgi) {
      multiplier *= 2;
      bonuses['흔들기'] = 'x2';
    }

    if (winner.goCount >= 1) {
      const goMult = Math.pow(2, Math.max(0, winner.goCount - 1));
      multiplier *= goMult;
      bonuses[`${winner.goCount}고`] = `x${goMult}`;
    }

    if (winner.bombs > 0) {
      multiplier *= Math.pow(2, winner.bombs);
      bonuses[`폭탄 ${winner.bombs}회`] = `x${Math.pow(2, winner.bombs)}`;
    }

    const loserPi = this.getPiCount(1 - winnerIdx);
    if (loserPi <= 6 && this.getPiCount(winnerIdx) >= 10) {
      multiplier *= 2;
      bonuses['피박'] = 'x2';
      await NotificationManager.showPiBak();
    }

    if (loser.captured.gwang.length === 0 && winner.captured.gwang.length >= 3) {
      multiplier *= 2;
      bonuses['광박'] = 'x2';
      await NotificationManager.showGwangBak();
    }

    if (multiplier > 1) {
      await NotificationManager.showMultiplier(multiplier);
    }

    const finalScore = baseScore * multiplier;
    const winnerName = winnerIdx === 0 ? '나' : (this.isMultiplayer ? '상대' : 'CPU');

    // v7: confetti on win, lose sound otherwise
    if (winnerIdx === 0) { soundManager.win(); launchConfetti(); } else soundManager.lose();

    this.recordStats(winnerIdx, finalScore, reason);
    this.showGameOver(winnerName, baseScore, finalScore, bonuses, reason);
  }

  recordStats(winnerIdx, finalScore, reason) {
    const won = winnerIdx === 0;
    const p = this.players[0];
    statistics.recordGame(won, won ? finalScore : 0, {
      bomb: p.bombs,
      ppuk: p.ppukCount,
      heundeulgi: p.heundeulgi ? 1 : 0,
      chongtong: reason === '총통' && winnerIdx === 0 ? 1 : 0,
      goCount: p.goCount,
    }, this.isMultiplayer);
  }

  showGameOver(winnerName, baseScore, finalScore, bonuses, reason) {
    const modal = document.getElementById('modal-container');
    let bonusHtml = '';
    Object.entries(bonuses).forEach(([k, v]) => {
      bonusHtml += `<div class="score-line"><span>${k}</span><span class="multiplier-badge">${v}</span></div>`;
    });

    modal.innerHTML = `
      <div class="game-over">
        <div class="game-over-content">
          <h2>${winnerName === '무승부' ? '🤝 무승부' : `🏆 ${winnerName} 승리!`}</h2>
          <p style="opacity:0.7">${reason}</p>
          <div class="score-panel">
            <div class="score-breakdown">
              ${baseScore > 0 ? `<div class="score-line"><span>기본 점수</span><span>${baseScore}점</span></div>` : ''}
              ${bonusHtml}
              ${finalScore > 0 ? `<div class="score-line score-total"><span>최종 점수</span><span>${finalScore}점</span></div>` : ''}
            </div>
          </div>
          <br>
          <button class="btn btn-primary" onclick="game.newGame(); document.getElementById('modal-container').innerHTML=''">새 게임</button>
          <button class="btn btn-secondary" onclick="backToMenu()">메뉴로</button>
        </div>
      </div>`;
  }

  // === RENDER ===
  render() {
    document.getElementById('deck-count').textContent = this.deck.length;
    document.getElementById('turn-count').textContent = this.turnCount;
    document.getElementById('turn-info').textContent =
      this.currentPlayer === 0 ? '👤 내 차례' :
      (this.isMultiplayer ? '🎮 상대 차례' : '🤖 CPU 차례');

    document.getElementById('player-area').classList.toggle('active', this.currentPlayer === 0);
    document.getElementById('cpu-area').classList.toggle('active', this.currentPlayer === 1);

    // v7: Score animation with pulse on change
    const pScoreEl = document.getElementById('player-score');
    const cScoreEl = document.getElementById('cpu-score');
    const newPScore = this.calculateScore(0);
    const newCScore = this.calculateScore(1);
    if (pScoreEl.dataset.prev && parseInt(pScoreEl.dataset.prev) !== newPScore) {
      pScoreEl.style.animation = 'none';
      pScoreEl.offsetHeight; // reflow
      pScoreEl.style.animation = 'scorePopup 0.3s ease-out';
    }
    pScoreEl.dataset.prev = newPScore;
    pScoreEl.textContent = newPScore;
    if (cScoreEl.dataset.prev && parseInt(cScoreEl.dataset.prev) !== newCScore) {
      cScoreEl.style.animation = 'none';
      cScoreEl.offsetHeight;
      cScoreEl.style.animation = 'scorePopup 0.3s ease-out';
    }
    cScoreEl.dataset.prev = newCScore;
    cScoreEl.textContent = newCScore;

    document.getElementById('player-go').textContent = this.players[0].goCount;
    document.getElementById('cpu-go').textContent = this.players[1].goCount;

    const badge = document.getElementById('player-heundeulgi-badge');
    badge.innerHTML = this.players[0].heundeulgi ?
      '<span class="heundeulgi-indicator">👋 흔들기</span>' : '';

    this.renderHand(0);
    this.renderCpuHand();
    this.renderFloor();
    this.renderCaptured(0);
    this.renderCaptured(1);
  }

  renderHand(playerIdx) {
    const el = document.getElementById('player-hand');
    el.innerHTML = '';
    this.players[playerIdx].hand.forEach(card => {
      const div = this.createCardEl(card, false);
      if (this.waitingForInput && this.currentPlayer === 0) {
        div.onclick = () => this.playCard(card.id);
      } else {
        div.classList.add('disabled');
      }
      el.appendChild(div);
    });
  }

  renderCpuHand() {
    const el = document.getElementById('cpu-hand');
    el.innerHTML = '';
    this.players[1].hand.forEach(() => {
      const div = document.createElement('div');
      div.className = 'card face-down';
      el.appendChild(div);
    });
  }

  renderFloor() {
    const el = document.getElementById('floor-cards');
    el.innerHTML = '';

    const groups = {};
    this.floor.forEach(c => {
      if (!groups[c.month]) groups[c.month] = [];
      groups[c.month].push(c);
    });

    Object.keys(groups).sort((a, b) => a - b).forEach(month => {
      groups[month].forEach(card => {
        const div = this.createCardEl(card, false);
        div.style.cursor = 'default';

        if (this.ppukStacks[card.month]) {
          const marker = document.createElement('div');
          marker.className = 'ppuk-marker';
          marker.textContent = '뻑';
          div.appendChild(marker);
        }

        el.appendChild(div);
      });
    });
  }

  renderCaptured(playerIdx) {
    const prefix = playerIdx === 0 ? 'player' : 'cpu';
    const p = this.players[playerIdx];

    ['gwang', 'tti', 'dan', 'pi'].forEach(type => {
      const el = document.getElementById(`${prefix}-${type}`);
      el.innerHTML = '';
      p.captured[type].forEach(card => {
        const div = this.createCardEl(card, true);
        el.appendChild(div);
      });
    });
  }

  // === ENHANCED CARD RENDERING WITH IMAGE SUPPORT (v7) ===
  createCardEl(card, small) {
    const div = document.createElement('div');
    div.className = `card ${card.type} ${small ? 'small' : ''} slide-in`;
    div.dataset.month = card.month;
    div.dataset.id = card.id;

    if (currentCardTheme !== 'emoji') {
      // Try to load image
      const cardIndex = ALL_CARDS.filter(c => c.month === card.month).indexOf(
        ALL_CARDS.find(c => c.id === card.id)
      );
      const mm = String(card.month).padStart(2, '0');
      const n = cardIndex + 1;
      const imgSrc = `assets/cards/${currentCardTheme}/${mm}_${n}.png`;

      const img = document.createElement('img');
      img.className = 'card-image';
      img.src = imgSrc;
      img.alt = `${card.monthName} ${card.name}`;
      img.onerror = function() {
        // Fallback to emoji display
        this.remove();
        div.innerHTML = `
          <div class="card-icon">${card.flower}</div>
          <div class="card-month">${card.month}</div>
          <div class="card-type">${card.icon}</div>
        `;
      };
      div.appendChild(img);
    } else {
      div.innerHTML = `
        <div class="card-icon">${card.flower}</div>
        <div class="card-month">${card.month}</div>
        <div class="card-type">${card.icon}</div>
      `;
    }
    return div;
  }
